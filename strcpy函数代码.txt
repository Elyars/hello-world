char* StrCpy(char* src)
{
	//计算src的长度len
	int len = sizeof(src) / sizeof(src[0]);
	//用malloc申请长度为len+1的空间
	char* p;
	p = (char*)malloc(len * sizeof(char));

	//将src字符逐个复制到新申请的内存中
	for (int i = 0; i <= len; i++)
	{
		
		if (i = len)
		{
			p = '\0';
		}
		else {
			stecpy(p, src[i]);
		}
	}

	return p;
}
